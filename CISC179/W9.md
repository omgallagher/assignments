```python
import tkinter as tk
from tkinter import messagebox

def calculate_mpg():
    try:
        gallons = float(entry_gallons.get())
        miles = float(entry_miles.get())
        if gallons <= 0:
            messagebox.showerror("Error", "Must input a number greater than 0 for gallons")
            return
        mpg = miles / gallons
        label_result.config(text=f"MPG: {mpg:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Input must be a number")

root = tk.Tk()
root.title("Gas mileage calculator")
root.geometry("300x200")

tk.Label(root, text="Number of gallons of gas the car holds:").pack(pady=5)
entry_gallons = tk.Entry(root)
entry_gallons.pack()

tk.Label(root, text="Number of miles car can drive on full tank:").pack(pady=5)
entry_miles = tk.Entry(root)
entry_miles.pack()

tk.Button(root, text="Calculate MPG", command=calculate_mpg).pack(pady=10)
label_result = tk.Label(root, text="MPG:")
label_result.pack(pady=10)

root.mainloop()
```
